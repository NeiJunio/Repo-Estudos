            SQL:

            CREATE TABLE `cashflowdb`.`expenses` (
                `Id` BIGINT NOT NULL AUTO_INCREMENT,
                `Title` VARCHAR(255) NOT NULL,
                `Description` VARCHAR(2000) NULL,
                `Date` DATETIME NOT NULL,
                `Amount` DECIMAL(10,2) NOT NULL,
                `PaymentType` INT NOT NULL,
                PRIMARY KEY (`Id`)
            );
